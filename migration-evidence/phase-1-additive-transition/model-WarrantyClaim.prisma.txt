model WarrantyClaim {
  id                     Int                             @id @default(autoincrement())
  branchId               Int
  stockItemId            Int
  supplierId             Int?
  saleReturnItemId       Int?
  previousClaimId        Int?
  claimNo                String
  status                 WarrantyClaimStatus             @default(DRAFT)
  reason                 String
  serviceProvider        String?
  externalClaimRef       String?
  trackingNumber         String?
  resolution             WarrantyClaimResolution?
  resolutionNote         String?
  replacementStockItemId Int?
  creditAmount           Decimal?                        @db.Decimal(12, 2)
  openedAt               DateTime                        @default(now())
  submittedAt            DateTime?
  providerReceivedAt     DateTime?
  resolvedAt             DateTime?
  cancelledAt            DateTime?
  createdByEmployeeId    Int?
  resolvedByEmployeeId   Int?
  createdAt              DateTime                        @default(now())
  updatedAt              DateTime                        @updatedAt
  branch                 Branch                          @relation(fields: [branchId], references: [id], onDelete: Restrict)
  stockItem              StockItem                       @relation(fields: [stockItemId], references: [id], onDelete: Restrict)
  supplier               Supplier?                       @relation(fields: [supplierId], references: [id], onDelete: Restrict)
  saleReturnItem         SaleReturnItem?                 @relation(fields: [saleReturnItemId], references: [id], onDelete: Restrict)
  previousClaim          WarrantyClaim?                  @relation("WarrantyClaimHistory", fields: [previousClaimId], references: [id], onDelete: Restrict)
  subsequentClaims       WarrantyClaim[]                 @relation("WarrantyClaimHistory")
  replacementStockItem   StockItem?                      @relation("WarrantyClaimReplacementStockItem", fields: [replacementStockItemId], references: [id], onDelete: Restrict)
  createdBy              EmployeeProfile?                @relation("WarrantyClaimCreatedBy", fields: [createdByEmployeeId], references: [id], onDelete: Restrict)
  resolvedBy             EmployeeProfile?                @relation("WarrantyClaimResolvedBy", fields: [resolvedByEmployeeId], references: [id], onDelete: Restrict)
  events                 WarrantyClaimEvent[]
  completionCommand      WarrantyClaimCompletionCommand?

  @@unique([branchId, claimNo])
  @@index([stockItemId, openedAt])
  @@index([supplierId, openedAt])
  @@index([saleReturnItemId])
  @@index([previousClaimId])
  @@index([replacementStockItemId])
  @@index([createdByEmployeeId])
  @@index([resolvedByEmployeeId])
  @@index([branchId, status, openedAt])
}
