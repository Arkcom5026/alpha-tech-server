model RepairJob {
  id               Int              @id @default(autoincrement())
  jobNo            String           @unique
  branchId         Int
  customerId       Int
  stockItemId      Int?
  deviceModel      String
  reportedSymptoms String
  technicianNotes  String?
  status           ServiceStatus    @default(RECEIVED)
  estimatedCost    Decimal          @default(0.0) @db.Decimal(12, 2)
  depositPaid      Decimal          @default(0.0) @db.Decimal(12, 2)
  technicianId     Int?
  createdAt        DateTime         @default(now())
  updatedAt        DateTime         @updatedAt
  branch           Branch           @relation(fields: [branchId], references: [id])
  customer         CustomerProfile  @relation(fields: [customerId], references: [id])
  stockItem        StockItem?       @relation(fields: [stockItemId], references: [id])
  technician       EmployeeProfile? @relation(fields: [technicianId], references: [id])
  partsUsed        RepairPartItem[]

  @@index([branchId])
  @@index([stockItemId])
}
